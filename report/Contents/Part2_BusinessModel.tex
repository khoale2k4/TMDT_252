\subsection{Chi tiết các thành phần trong mô hình}

\begin{enumerate}
    \item \textbf{Phân khúc khách hàng (Customer Segments):}
    Đây là cốt lõi của mô hình Sàn giao dịch hai chiều. Cần phải xác định rõ khách hàng mục tiêu. Cần phục vụ ai, họ cần gì, mình có thể làm gì cho họ.
    \begin{itemize}
        \item \textbf{B2B (Chủ sân/Người quản lý):} Các chủ sở hữu cụm sân thể thao cần công cụ quản lý chuyên nghiệp, tự động hóa và báo cáo tài chính chính xác.
        \item \textbf{B2C (Người chơi thể thao):} Cá nhân hoặc nhóm người chơi cần tìm kiếm và đặt sân nhanh chóng, minh bạch về giá cả và tiện ích.
    \end{itemize}

    \item \textbf{Giải pháp giá trị (Value Propositions):}Thách thức lớn nhất là bài toán "con gà - quả trứng": Không có sân thì người chơi không dùng, không có người chơi thì chủ sân không tham gia.
    \begin{itemize}
        \item \textbf{Cho chủ sân:} "Không thu phí hoa hồng" và cung cấp hệ thống quản lý thay thế giấy/Excel biến CourtMate thành một phần mềm SaaS tiện lợi. Giảm thiểu sai sót thủ công, tối ưu hóa công suất khai thác sân (đặc biệt là giờ thấp điểm), quản lý tài chính tích hợp hóa đơn điện tử.
        \item \textbf{Cho người chơi:}Từ dữ liệu sân trống của chủ sân,  tạo ra giá trị "Tiết kiệm" và "Thuận tiện" cho người chơi. Trải nghiệm đặt sân nhanh chóng; hiển thị trạng thái sân thời gian thực; tích hợp thanh toán linh hoạt; hệ thống Matchmaking tự động ghép trận và tính năng Recurring Booking cho phép đặt lịch cố định dài hạn.
    \end{itemize}

    \item \textbf{Các kênh truyền thông và phân phối (Channels):} Sử dụng các kênh chi phí thấp/lan truyền nhanh để kéo khách đến, và dùng sự tự động hóa cùng uy tín cộng đồng để giữ khách ở lại.
    \begin{itemize}
        \item Ứng dụng Web/Mobile cho người chơi. Đồng thời hợp tác trực tiếp, hướng dẫn chủ sân số hóa quy trình.
        \item Bảng điều khiển (Admin Panel) chuyên dụng cho chủ sân.
        \item Đẩy mạnh qua KOLs thể thao và các hội nhóm. Thể thao là bộ môn có tính cộng đồng cao
    \end{itemize}

    \item \textbf{Quan hệ khách hàng (Customer Relationships):} Là một mắt xích quan trọng. Chủ sân, người chơi có thể hài lòng với các chức năng, sự tiện lợi ứng dụng mang lại. Nhưng chỉ một vài điều nhỏ nhặt, một chút khó chịu, sự cố trong khi dùng ứng dụng cũng đủ để khách hàng khó chịu và rời đi.
    \begin{itemize}
        \item Đưa ra giải pháp hỗ trợ kỹ thuật 24/7 cho các chủ sân SaaS. Cung cấp sự sẵn sàng, giúp giải đáp mọi thắc mắc, vấn đề trong thời gian ngắn cho khách hàng.
        \item Hệ thống đánh giá và phản hồi đa chiều giữa người chơi và chủ sân. 
    \end{itemize}

    \item \textbf{Dòng doanh thu (Revenue Streams):}
    \begin{itemize}
        \item \textbf{B2B (SaaS):} Phí thuê bao hàng tháng từ chủ sân với cam kết 0\% hoa hồng, giúp họ tối ưu hóa lợi nhuận biên.
        \item \textbf{B2C (Premium):} Gói hội viên cao cấp Premium cung cấp các đặc quyền tạo, tăng trải nghiệm người dùng, giúp khách hàng có thể sử dụng nhanh, thuận tiện hơn. Ví dụ: Account Premium được quyền đặt sân trước 7 ngày (so với 3 ngày của account thường)
        \item \textbf{B2B2C (Ads):} Doanh thu quảng cáo từ các thực thể thương mại và nhãn hàng thể thao tài trợ. Quảng cáo không chỉ là treo banner. Dựa trên dữ liệu hành vi, cung cấp Quảng cáo nhắm mục tiêu cho các nhãn hàng thể thao với tỷ lệ chuyển đổi cao.
    \end{itemize}

    \item \textbf{Các hoạt động chính (Key Activities):} Cốt lõi không chỉ là duy trì một ứng dụng thông thường, mà là vận hành một hệ thống phức tạp phục vụ cùng lúc hai luồng khách hàng (B2B và B2C) và liên tục tối ưu hóa luồng dữ liệu
    \begin{itemize}
        \item Phát triển và bảo trì nền tảng kỹ thuật. Đảm bảo kiến trúc hệ thống (phân tách rõ luồng dịch vụ cho chủ sân, người chơi và lõi AI) luôn ổn định để xử lý giao dịch và cập nhật giá động theo thời gian thực.
        \item Kết nối và mở rộng mạng lưới các đối tác cụm sân. Càng kéo được nhiều sân lên hệ thống, hiệu ứng mạng lưới càng mạnh.
        \item Tiếp thị và chăm sóc khách hàng. Tập trung vào việc thay đổi thói quen người dùng từ đặt sân truyền thống (gọi điện, nhắn tin) sang tự động hóa hoàn toàn.
    \end{itemize}

    \item \textbf{Các nguồn lực chính (Key Resources):} Tài sản quý giá nhất của mô hình này nằm ở năng lực xử lý kỹ thuật của nền tảng và khối lượng dữ liệu độc quyền thu thập được
    \begin{itemize}
        \item Đội ngũ kỹ sư phát triển phần mềm.
        \item Hạ tầng máy chủ Cloud và cơ sở dữ liệu.
        \item Dữ liệu về bản đồ và vị trí các cụm sân.
    \end{itemize}

    \item \textbf{Các đối tác chính (Key Partners):} Tận dụng thế mạnh của các bên thứ ba để tối ưu chi phí phát triển, từ đó dồn toàn lực vào các tính năng lõi mang lại lợi thế cạnh tranh.
    \begin{itemize}
        \item \textbf{Thanh toán:} VietQR xử lý giao dịch tức thời và bảo mật. Điều này giúp giảm thiểu rủi ro "bùng sân" và tạo luồng tiền mặt xoay vòng nhanh, minh bạch cho chủ sân.
        \item \textbf{Công nghệ \& Vận hành:} MISA meInvoice (hóa đơn điện tử giải quyết trọn vẹn bài toán xuất hóa đơn, tăng thêm giá trị cho phần mềm quản lý
        \item \textbf{Thương mại:} Các nhãn hàng thể thao tài trợ và quảng cáo trên nền tảng đối tác mang lại dòng doanh thu trọng yếu, bù đắp trực tiếp cho chiến lược "Không thu phí hoa hồng".
    \end{itemize}

    \item \textbf{Cơ cấu chi phí (Cost Structure):}
    \begin{itemize}
        \item Chi phí vận hành hạ tầng Cloud (AWS/Azure). Đặc biệt là chi phí duy trì các cụm dịch vụ, container và cơ sở dữ liệu vận hành 24/7 để phục vụ AI Pricing và quản lý booking.
        \item Phí sử dụng API Google Maps.
        \item  Khoản đầu tư lớn nhất ở giai đoạn đầu là chi phí nhân sự và Marketing. Các chi phí này nhằm mục đích đó chính là lôi kéo cả chủ sân (bằng đội ngũ sale trực tiếp) và người chơi (bằng ads/khuyến mãi) gia nhập hệ sinh thái.
    \end{itemize}
\end{enumerate}
